### vant按需加载：
cnpm install babel-plugin-import -D
在babel.config.js中配置：
module.exports = {
  presets: [
    '@babel/preset-env'
  ],
  plugins: [
    ['import', {
      libraryName: 'vant',
      libraryDirectory: 'es',
      style: true
    }, 'vant']
  ]
};

直接使用：
import Vue from 'vue';
import { Button } from 'vant';

Vue.use(Button);
